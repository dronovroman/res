            const petal = new THREE.Mesh(petalGeometry, petalMaterial);
            petal.position.y = 0.3; // Position at the top of the stem
            petal.castShadow = true;
            flowerGroup.add(petal);

            flowerGroup.position.set(x, y, z);
            flowerGroup.scale.setScalar(Math.random() * 0.5 + 0.7); // Random size
            scene.add(flowerGroup);
        }


        function createSky() {
            // Sun
            const sunGeometry = new THREE.SphereGeometry(20, 32, 32);
            const sunMaterial = new THREE.MeshBasicMaterial({
                color: 0xffddaa, // Warm yellow
                fog: false // Sun shouldn't be affected by fog
            });
            const sun = new THREE.Mesh(sunGeometry, sunMaterial);
            sun.position.set(-150, 150, -200); // Position the sun
            scene.add(sun);

            // Clouds using simple geometry
             for (let i = 0; i < 15; i++) {
                createCloud(
                    Math.random() * 300 - 150, // x
                    80 + Math.random() * 40,  // y (height)
                    Math.random() * 300 - 150, // z
                    Math.random() * 10 + 8    // size
                );
            }
        }

        function createCloud(x, y, z, size) {
            const cloudGroup = new THREE.Group();
            const cloudMaterial = new THREE.MeshBasicMaterial({
                color: 0xffffff,
                transparent: true,
                opacity: 0.85,
                depthWrite: false // Helps with transparency sorting issues
            });

            const numSpheres = Math.floor(Math.random() * 6) + 4; // 4-9 spheres per cloud
            for (let i = 0; i < numSpheres; i++) {
                const sphereSize = size * (0.4 + Math.random() * 0.6);
                const sphereGeometry = new THREE.SphereGeometry(sphereSize, 8, 6); // Low poly spheres
                const sphere = new THREE.Mesh(sphereGeometry, cloudMaterial);

                sphere.position.set(
                    (Math.random() - 0.5) * size * 2.5,
                    (Math.random() - 0.5) * size * 0.8,
                    (Math.random() - 0.5) * size * 1.5
                );
                sphere.rotation.set(Math.random() * Math.PI, Math.random() * Math.PI, Math.random() * Math.PI);
                cloudGroup.add(sphere);
            }

            cloudGroup.position.set(x, y, z);
            scene.add(cloudGroup);

            // Add simple movement logic
            cloudGroup.userData = {
                isCloud: true, // Flag for identification
                speedX: (Math.random() - 0.5) * 0.01 + 0.01, // Slow horizontal drift
                speedZ: (Math.random() - 0.5) * 0.005,
                update: function(dt) {
                    cloudGroup.position.x += this.speedX * dt * 60; // Scale by approx frame rate
                    cloudGroup.position.z += this.speedZ * dt * 60;
                    // Wrap around screen edges
                    if (cloudGroup.position.x > 200) cloudGroup.position.x = -200;
                    if (cloudGroup.position.x < -200) cloudGroup.position.x = 200;
                    if (cloudGroup.position.z > 200) cloudGroup.position.z = -200;
                    if (cloudGroup.position.z < -200) cloudGroup.position.z = 200;
                }
            };
            return cloudGroup;
        }

        function createTrees() {
            trees = []; // Clear existing trees if restarting
            const numTrees = 20;
            const spawnRadiusMin = 20;
            const spawnRadiusMax = 80;

            for (let i = 0; i < numTrees; i++) {
                const angle = Math.random() * Math.PI * 2;
                const radius = spawnRadiusMin + Math.random() * (spawnRadiusMax - spawnRadiusMin);
                const x = Math.sin(angle) * radius + (Math.random() - 0.5) * 10; // Add some jitter
                const z = Math.cos(angle) * radius + (Math.random() - 0.5) * 10;

                 const groundY = (Math.sin(x * 0.05) * Math.cos(z * 0.05)) * 1.5 + Math.random() * 0.1; // Approx height

                const tree = createTree(x, groundY, z);
                trees.push(tree);
            }
        }

        function createTree(x, y, z) {
            const treeGroup = new THREE.Group();
             treeGroup.userData.isTree = true; // Flag for identification
            const scale = 1.0 + Math.random() * 0.5; // Randomize size

            // Trunk
            const trunkHeight = 6 + Math.random() * 3;
            const trunkRadius = 0.5 + Math.random() * 0.3;
            const trunkGeometry = new THREE.CylinderGeometry(trunkRadius * 0.8, trunkRadius, trunkHeight, 8); // Tapered
            const trunkMaterial = new THREE.MeshStandardMaterial({
                color: 0x8B4513, // Saddle Brown
                roughness: 0.9,
                metalness: 0.1
            });
            const trunk = new THREE.Mesh(trunkGeometry, trunkMaterial);
            trunk.position.y = trunkHeight / 2; // Base of trunk at y=0
            trunk.castShadow = true;
            trunk.receiveShadow = true;
            treeGroup.add(trunk);

            // Leaves
            const leavesMaterial = new THREE.MeshStandardMaterial({
                color: 0x228B22, // Forest Green
                roughness: 0.8,
                metalness: 0.1
            });

            const leafLevels = 3 + Math.floor(Math.random() * 3);
            let currentY = trunkHeight * 0.7;
            let currentRadius = trunkRadius * 5 + Math.random() * 2;

            for(let i = 0; i < leafLevels; i++) {
                const numBlobs = 5 + Math.floor(Math.random() * 5);
                for(let j = 0; j < numBlobs; j++) {
                    const blobSize = currentRadius * (0.3 + Math.random() * 0.5);
                    const blobGeometry = new THREE.IcosahedronGeometry(blobSize, 0); // Low poly
                    const blob = new THREE.Mesh(blobGeometry, leavesMaterial);
                    blob.position.set(
                        (Math.random() - 0.5) * currentRadius * 0.8,
                        currentY + (Math.random() - 0.3) * blobSize * 1.5, // Vertical variation
                        (Math.random() - 0.5) * currentRadius * 0.8
                    );
                    blob.castShadow = true;
                    blob.receiveShadow = true;
                    treeGroup.add(blob);
                }
                 currentY += currentRadius * 0.5;
                 currentRadius *= 0.7;
            }


            treeGroup.scale.set(scale, scale, scale);
            treeGroup.rotation.y = Math.random() * Math.PI * 2;
            treeGroup.position.set(x, y, z);
            scene.add(treeGroup);

            treeGroup.userData.boundingBox = new THREE.Box3().setFromObject(trunk);

            return treeGroup;
        }


        function createRabbit(treeIndex) {
            if (treeIndex < 0 || treeIndex >= trees.length) {
                 console.error("Invalid tree index for rabbit spawn:", treeIndex);
                 return null;
            }
            const tree = trees[treeIndex];
            const treePos = tree.position.clone();
            const treeScale = tree.scale.x; // Assuming uniform scale

            // --- Rabbit Geometry ---
            const rabbitGroup = new THREE.Group();
            const rabbitMaterial = new THREE.MeshStandardMaterial({
                color: 0xEAE0D5, // Off-white / beige
                roughness: 0.8,
                metalness: 0.1
            });
            const pinkMaterial = new THREE.MeshStandardMaterial({ color: 0xFFC0CB, roughness: 0.9 });
            const blackMaterial = new THREE.MeshStandardMaterial({ color: 0x111111, roughness: 0.5 });

            // Body
            const bodyGeo = new THREE.SphereGeometry(0.5, 12, 8);
            bodyGeo.scale(1, 0.8, 1.2);
            const body = new THREE.Mesh(bodyGeo, rabbitMaterial);
            body.position.y = 0.4;
            body.castShadow = true;
            rabbitGroup.add(body);

            // Head
            const headGeo = new THREE.SphereGeometry(0.35, 12, 8);
            const head = new THREE.Mesh(headGeo, rabbitMaterial);
            head.position.set(0, 0.8, 0.4);
            head.castShadow = true;
            rabbitGroup.add(head);

            // Ears
            const earHeight = 0.7;
            const earRadius = 0.08;
            const earGeo = new THREE.CylinderGeometry(earRadius, earRadius * 0.7, earHeight, 6);
            earGeo.translate(0, earHeight / 2, 0);
            const posAttr = earGeo.attributes.position;
            for (let i = 0; i < posAttr.count; i++) {
                const y = posAttr.getY(i);
                const bendFactor = Math.sin((y / earHeight) * Math.PI / 2);
                posAttr.setX(i, posAttr.getX(i) + bendFactor * 0.1);
            }
            earGeo.computeVertexNormals();


            const leftEar = new THREE.Mesh(earGeo, rabbitMaterial);
            leftEar.position.set(-0.15, 1.05, 0.35);
            leftEar.rotation.set(-Math.PI / 10, Math.PI / 16, Math.PI / 12);
            leftEar.castShadow = true;
            rabbitGroup.add(leftEar);

            const rightEar = leftEar.clone();
            rightEar.position.x = 0.15;
            rightEar.rotation.z *= -1;
            rabbitGroup.add(rightEar);

            const innerEarGeo = new THREE.CylinderGeometry(earRadius * 0.6, earRadius * 0.4, earHeight * 0.8, 6);
            innerEarGeo.translate(0, earHeight / 2 * 0.8, 0.01);
            const leftInner = new THREE.Mesh(innerEarGeo, pinkMaterial);
            leftEar.add(leftInner);
            const rightInner = leftInner.clone();
            rightEar.add(rightInner);

            // Eyes
            const eyeGeo = new THREE.SphereGeometry(0.05, 6, 4);
            const leftEye = new THREE.Mesh(eyeGeo, blackMaterial);
            leftEye.position.set(-0.12, 0.9, 0.7);
            head.add(leftEye);

            const rightEye = leftEye.clone();
            rightEye.position.x = 0.12;
            head.add(rightEye);

            // Nose
            const noseGeo = new THREE.SphereGeometry(0.06, 6, 4);
            const nose = new THREE.Mesh(noseGeo, pinkMaterial);
            nose.position.set(0, 0.8, 0.72);
            head.add(nose);

            // Tail
            const tailGeo = new THREE.SphereGeometry(0.2, 8, 6);
            const tail = new THREE.Mesh(tailGeo, new THREE.MeshStandardMaterial({ color: 0xffffff, roughness: 0.9 }));
            tail.position.set(0, 0.3, -0.55);
            body.add(tail);
            tail.castShadow = true;

            // --- Positioning ---
            const treeRadiusEstimate = (tree.userData.boundingBox.max.x - tree.userData.boundingBox.min.x) * treeScale * 0.6;
            const distanceFromTrunk = treeRadiusEstimate + 0.2 + Math.random() * 0.5;
            const angleRelativeToTree = Math.random() * Math.PI * 1.5 - Math.PI * 0.75;

            const rabbitX = treePos.x + Math.cos(angleRelativeToTree) * distanceFromTrunk;
            const rabbitZ = treePos.z + Math.sin(angleRelativeToTree) * distanceFromTrunk;
            const rabbitGroundY = (Math.sin(rabbitX * 0.05) * Math.cos(rabbitZ * 0.05)) * 1.5 + Math.random() * 0.1;

            rabbitGroup.position.set(rabbitX, rabbitGroundY, rabbitZ);
            const lookTarget = new THREE.Vector3( camera.position.x * 0.5, rabbitGroundY + 1.0, camera.position.z * 0.5 );
            rabbitGroup.lookAt(lookTarget);
            rabbitGroup.rotation.y += (Math.random() - 0.5) * 0.4;


            scene.add(rabbitGroup);

            // --- Game Logic Data ---
            rabbitGroup.userData = {
                isRabbit: true, // Flag to identify rabbit groups
                isShot: false,
                appearTime: clock.getElapsedTime(),
                stayDuration: 1.0 + Math.random() * 2.5, // 1-2 seconds visible
                associatedName: null,
                // associatedText: null, // Removed 3D text
                treeIndex: treeIndex,
                initialY: rabbitGroundY,
                bobblePhase: Math.random() * Math.PI * 2,
                update: function(dt) {
                    const currentTime = clock.getElapsedTime();
                    if (this.isShot) {
                        // Falling animation
                        if (rabbitGroup.position.y > this.initialY - 1.5) { // Fall below ground level
                            rabbitGroup.position.y -= dt * 3; // Fall speed
                            rabbitGroup.rotation.x += dt * 5; // Tumble
                            rabbitGroup.rotation.z += dt * 2;
                        }
                        // NO NEED TO UPDATE associatedText position anymore
                    } else {
                        // Bobble animation
                        const bobbleAmount = Math.sin(currentTime * 4 + this.bobblePhase) * 0.05;
                        rabbitGroup.position.y = this.initialY + bobbleAmount;

                        // Check if rabbit should disappear
                        if (currentTime - this.appearTime > this.stayDuration) {
                            scene.remove(rabbitGroup);
                            const index = rabbits.findIndex(r => r === rabbitGroup);
                            if (index > -1) rabbits.splice(index, 1);
                            // NO NEED TO REMOVE associatedText anymore
                        }
                    }
                }
            };
            return rabbitGroup;
        }

        function setupCrosshair() {
            crosshair = document.getElementById('crosshair');
            crosshair.style.display = 'block'; // Make it visible

            updateCrosshairPosition({ clientX: window.innerWidth / 2, clientY: window.innerHeight / 2 });

            gameContainer.addEventListener('mousemove', updateCrosshairPosition);
            gameContainer.addEventListener('click', handleShoot);

            gameContainer.addEventListener('click', () => {
                if (gameActive && document.pointerLockElement !== gameContainer) {
                    gameContainer.requestPointerLock = gameContainer.requestPointerLock || gameContainer.mozRequestPointerLock;
                    gameContainer.requestPointerLock();
                }
            });

            document.addEventListener('pointerlockchange', lockChangeAlert, false);
            document.addEventListener('mozpointerlockchange', lockChangeAlert, false);
        }

        function lockChangeAlert() {
            if (document.pointerLockElement === gameContainer || document.mozPointerLockElement === gameContainer) {
                console.log('Pointer Lock active.');
                 gameContainer.addEventListener('mousemove', updateCrosshairPointerLock, true);
            } else {
                console.log('Pointer Lock deactivated.');
                gameContainer.removeEventListener('mousemove', updateCrosshairPointerLock, true);
            }
        }

        function updateCrosshairPosition(event) {
             if (!gameActive || document.pointerLockElement === gameContainer) return;

            const rect = gameContainer.getBoundingClientRect();
            const x = event.clientX - rect.left;
            const y = event.clientY - rect.top;

            crosshair.style.left = x + 'px';
            crosshair.style.top = y + 'px';

            mouse.x = (x / rect.width) * 2 - 1;
            mouse.y = -(y / rect.height) * 2 + 1;
        }

        function updateCrosshairPointerLock(event) {
             if (!gameActive || document.pointerLockElement !== gameContainer) return;

            const rect = gameContainer.getBoundingClientRect();
            let currentX = parseFloat(crosshair.style.left || rect.width / 2);
            let currentY = parseFloat(crosshair.style.top || rect.height / 2);

            currentX += event.movementX;
            currentY += event.movementY;

            currentX = Math.max(0, Math.min(rect.width, currentX));
            currentY = Math.max(0, Math.min(rect.height, currentY));

            crosshair.style.left = currentX + 'px';
            crosshair.style.top = currentY + 'px';

            mouse.x = (currentX / rect.width) * 2 - 1;
            mouse.y = -(currentY / rect.height) * 2 + 1;
        }


        function handleShoot(event) {
            if (!gameActive || !shootable || nameSet.length === 0) return;

            playGunshot();
            raycaster.setFromCamera(mouse, camera);

            const rabbitMeshes = [];
             scene.traverse(obj => {
                if (obj.isGroup && obj.userData.isRabbit) {
                     obj.traverse(child => { if (child.isMesh) rabbitMeshes.push(child); });
                }
            });

            const intersects = raycaster.intersectObjects(rabbitMeshes, false);

            if (intersects.length > 0) {
                let hitObject = intersects[0].object;
                let rabbitGroup = null;
                while (hitObject) {
                    if (hitObject.isGroup && hitObject.userData.isRabbit) {
                        rabbitGroup = hitObject;
                        break;
                    }
                     if (!hitObject.parent || hitObject === scene) break;
                    hitObject = hitObject.parent;
                }

                if (rabbitGroup && !rabbitGroup.userData.isShot) {
                     console.log("Rabbit Hit!", rabbitGroup);
                    shootRabbit(rabbitGroup);
                    playHitSound();
                } else {
                     console.log("Missed or hit already shot rabbit", rabbitGroup);
                }
            } else {
                 console.log("Missed");
            }

            shootable = false;
            setTimeout(() => { shootable = true; }, 250); // Slightly faster shooting allowed
        }


        function playSound(type) {
            if (!audioContext) return;
            let oscillator, gainNode, frequency, duration, volume, decayTime;

            switch (type) {
                case 'gunshot':
                    oscillator = audioContext.createOscillator();
                    gainNode = audioContext.createGain();
                    frequency = 120; duration = 0.15; volume = 0.4; decayTime = 0.1;
                    oscillator.type = 'sawtooth';
                    oscillator.frequency.setValueAtTime(frequency, audioContext.currentTime);
                    oscillator.frequency.exponentialRampToValueAtTime(frequency * 0.5, audioContext.currentTime + duration * 0.5);
                    gainNode.gain.setValueAtTime(volume, audioContext.currentTime);
                    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + decayTime);
                    oscillator.connect(gainNode); gainNode.connect(audioContext.destination);
                    oscillator.start(audioContext.currentTime); oscillator.stop(audioContext.currentTime + duration);

                    const noiseBuffer = audioContext.createBuffer(1, audioContext.sampleRate * 0.1, audioContext.sampleRate);
                    const output = noiseBuffer.getChannelData(0);
                    for (let i = 0; i < output.length; i++) output[i] = Math.random() * 2 - 1;
                    const noiseSource = audioContext.createBufferSource(); noiseSource.buffer = noiseBuffer;
                    const noiseGain = audioContext.createGain();
                    noiseGain.gain.setValueAtTime(0.2, audioContext.currentTime);
                    noiseGain.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.05);
                    noiseSource.connect(noiseGain); noiseGain.connect(audioContext.destination);
                    noiseSource.start(audioContext.currentTime); noiseSource.stop(audioContext.currentTime + 0.1);
                    break;
                case 'hit':
                    oscillator = audioContext.createOscillator();
                    gainNode = audioContext.createGain();
                    frequency = 900; duration = 0.2; volume = 0.3; decayTime = 0.15;
                    oscillator.type = 'sine';
                    oscillator.frequency.setValueAtTime(frequency, audioContext.currentTime);
                    gainNode.gain.setValueAtTime(volume, audioContext.currentTime);
                    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + decayTime);
                    oscillator.connect(gainNode); gainNode.connect(audioContext.destination);
                    oscillator.start(audioContext.currentTime); oscillator.stop(audioContext.currentTime + duration);
                    break;
            }
        }
        function playGunshot() { playSound('gunshot'); }
        function playHitSound() { playSound('hit'); }


         function shootRabbit(rabbit) {
            if (!rabbit || !rabbit.userData || rabbit.userData.isShot) return;

            rabbit.userData.isShot = true;

            if (nameSet.length > 0) {
                const nameIndex = Math.floor(Math.random() * nameSet.length);
                const name = nameSet[nameIndex];
                nameSet.splice(nameIndex, 1);
                usedNames.push(name); // Keep track for final screen

                rabbit.userData.associatedName = name; // Store name with rabbit data if needed later

                // --- Update the on-screen list ---
                const listItem = document.createElement('li');
                listItem.textContent = name;
                shotNamesListElement.appendChild(listItem);
                // Scroll to bottom of list if it overflows
                const listContainer = document.getElementById('shot-names-display');
                 listContainer.scrollTop = listContainer.scrollHeight;
                // ---------------------------------

                // Update score display
                document.getElementById('names-left').textContent = nameSet.length;

                // Check for game over condition
                if (nameSet.length === 0) {
                    console.log("All names used! Ending game soon.");
                    setTimeout(endGame, 3000); // Shortened delay slightly
                }
            } else {
                 console.warn("Shot a rabbit, but no names left in nameSet.");
            }
        }

        function spawnRabbit() {
            if (!gameActive || nameSet.length === 0 || trees.length === 0) return;

             // Limit concurrent rabbits
            if (rabbits.length >= 5) { // Allow slightly more rabbits on screen
                 // console.log("Max rabbits reached, waiting...");
                 setTimeout(spawnRabbit, 800); // Check again sooner
                 return;
            }

            let attempts = 0;
            let chosenTreeIndex = -1;
            while (attempts < trees.length * 2) {
                 let potentialIndex = Math.floor(Math.random() * trees.length);
                 // Basic check is enough now, don't need complex free check
                 chosenTreeIndex = potentialIndex;
                 break;
                 attempts++;
            }


            if (chosenTreeIndex !== -1) {
                const newRabbit = createRabbit(chosenTreeIndex);
                if (newRabbit) {
                    rabbits.push(newRabbit);
                    // console.log(`Spawned rabbit at tree ${chosenTreeIndex}. Total rabbits: ${rabbits.length}`);
                } else {
                     console.error("Failed to create rabbit.");
                }
            } else {
                console.log("Could not find a tree to spawn rabbit (shouldn't happen with current logic).");
            }


            // Schedule the next spawn attempt - MORE FREQUENTLY
            const spawnDelay = 500 + Math.random() * 1000; // 0.5 - 1.5 seconds delay
            setTimeout(spawnRabbit, spawnDelay);
        }

        function startGame() {
            // 1. Get names
            const nameInputs = document.querySelectorAll('.name-input');
            nameSet = [];
            usedNames = [];
            for (let i = 0; i < nameInputs.length; i++) {
                const name = nameInputs[i].value.trim();
                nameSet.push(name || `Player ${i + 1}`);
            }

            if (nameSet.length === 0) {
                 alert("Please enter at least one name!");
                 return;
            }

            // 2. Hide Start Screen, Show Game UI
            document.getElementById('start-screen').style.display = 'none';
            document.getElementById('score-display').style.display = 'block';
            document.getElementById('shot-names-display').style.display = 'block'; // Show the list container
            shotNamesListElement.innerHTML = ''; // Clear the list content
            document.getElementById('names-left').textContent = nameSet.length;


            // 3. Setup Scene Elements
            cleanUpScene();
            createTrees();

            // 4. Reset Game State Variables
            rabbits = [];
            // nameTexts = []; // No longer needed
            shootable = true;
            gameActive = true;
            if (!clock.running) clock.start(); // Ensure clock is running

            // 5. Setup Controls
            setupCrosshair();

            // 6. Start Game Logic
            console.log("Game Starting with names:", nameSet);
            spawnRabbit();
        }

        function endGame() {
            console.log("Game Ending.");
            gameActive = false;

            if (document.exitPointerLock) document.exitPointerLock();
            else if (document.mozExitPointerLock) document.mozExitPointerLock();

            // Hide Game UI, Show End Screen

            document.getElementById('shot-names-display').style.display = 'none'; // Hide in-game list

            document.removeEventListener('pointerlockchange', lockChangeAlert, false);
            document.removeEventListener('mozpointerlockchange', lockChangeAlert, false);

            // Display final results
            const resultsList = document.getElementById('results-list');
            resultsList.innerHTML = ''; // Clear previous results
            usedNames.forEach((name) => { // Index not needed as list provides numbering
                const li = document.createElement('li');
                li.textContent = `${name}`;
                resultsList.appendChild(li);
            });

            // clock.stop(); // Optional
        }

         function cleanUpScene() {

                    objectsToRemove.push(obj);

                  // Remove lights? Only if necessary, usually they can stay.

                  //     else obj.material.dispose();

        }

        function animate() {
            requestAnimationFrame(animate);
            const dt = clock.getDelta();

            if (gameActive) {
                 for (let i = rabbits.length - 1; i >= 0; i--) {
                    const rabbit = rabbits[i];
                    // Check parent just in case it was removed externally somehow
                    if (rabbit.parent && rabbit.userData && rabbit.userData.update) {
                        rabbit.userData.update(dt);
                    } else if (!rabbit.parent) {
                        // Rabbit was already removed, clean up array
                        rabbits.splice(i, 1);
                    }
                 }
                // NO NEED TO UPDATE nameTexts anymore
            }

            // Update clouds
            scene.traverse(child => {
                if (child.userData && child.userData.isCloud && child.userData.update) {
                    child.userData.update(dt);
                }
            });

            renderer.render(scene, camera);
        }
